#include <iostream>

double func(double x) {
    return 2 * pow(x, 2) - 2 * x + 5 / 2;
}

int FibonachiX(int x) {
    if (x <= 0) return 0;
    if (x <= 1) return 1;

    int prev = 0;
    int curr = 1;
    int i = 0;

    while (curr < x) {
        int next = prev + curr;
        prev = curr;
        curr = next;
        i++;
    }

    return i;
}

float FibonachiN(int n) {
    if (n <= 1) return n;

    int prev = 0, curr = 1;
    for (int i = 2; i <= n; ++i) {
        int next = prev + curr;
        prev = curr;
        curr = next;
    }
    return curr;
}
